CREATE TABLE `blog_cms_user` (
  `USER_ID` varchar(30) NOT NULL,
  `USER_NAME` varchar(55) DEFAULT NULL,
  `USER_EN_NAME` varchar(100) DEFAULT NULL,
  `LOGIN_NAME` varchar(50) DEFAULT NULL,
  `USER_PASSWORD` varchar(255) DEFAULT NULL,
  `SEX` decimal(1,0) DEFAULT NULL,
  `BRIEF` varchar(1024) DEFAULT NULL,
  `BIRTH_DATE` date DEFAULT NULL,
  `POSITION_COUNTRY` varchar(100) DEFAULT NULL,
  `POSITION_PROVINCE` varchar(100) DEFAULT NULL,
  `POSITION_CITY` varchar(100) DEFAULT NULL,
  `POSITION_OTHER_DETAIL` varchar(255) DEFAULT NULL,
  `MOBILE_PHONE` varchar(50) DEFAULT NULL,
  `WORK_PHONE` varchar(50) DEFAULT NULL,
  `STATUS_FLAG` decimal(1,0) DEFAULT '0' COMMENT '冻结状态 0 未冻结 1 已冻结',
  `CREATE_TIME` datetime DEFAULT NULL,
  `DELETE_FLAG` decimal(1,0) DEFAULT '0' COMMENT '删除状态 0未删除 1已删除',
  PRIMARY KEY (`USER_ID`),
  UNIQUE KEY `loginName` (`LOGIN_NAME`) COMMENT '登录名唯一'
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


CREATE TABLE `blog_cms_menu` (
  `MENU_ID` varchar(30) NOT NULL,
  `MENU_KEY` varchar(50) DEFAULT NULL,
  `MENU_NAME` varchar(50) DEFAULT NULL,
  `MENU_URL` varchar(200) DEFAULT NULL,
  `OPEN_TYPE` decimal(1,0) DEFAULT '0' COMMENT '0不打开新的页面，1打开新的页面',
  `DELETE_FLG` decimal(1,0) DEFAULT '0' COMMENT '0正常1删除',
  `PARENT_MENU_ID` varchar(30) DEFAULT NULL,
  `ORDER_NUM` int(11) DEFAULT NULL,
  `MENU_CREATE_ID` varchar(30) DEFAULT NULL,
  `MENU_CREATE_TIME` datetime DEFAULT NULL,
  `MENU_LASTMODIFY_ID` varchar(30) DEFAULT NULL,
  `MENU_LASTMODIFY_TIME` datetime DEFAULT NULL,
  `MENU_CSS_CLASS` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`MENU_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `blog_cms_menu_css_class` (
  `MENU_CSS_CLASS` varchar(50) NOT NULL,
  `MENU_NAME` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`MENU_CSS_CLASS`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `blog_cms_role` (
  `ROLE_ID` varchar(30) NOT NULL,
  `ROLE_NAME` varchar(100) DEFAULT NULL,
  `ROLE_EN_NAME` varchar(100) DEFAULT NULL,
  `ROLE_BRIEF` varchar(255) DEFAULT NULL,
  `ROLE_CREATE_TIME` datetime DEFAULT NULL,
  `ROLE_CREATE_ID` varchar(30) DEFAULT NULL,
  `ROLE_STATE_FLAG` decimal(1,0) DEFAULT '0' COMMENT '0正常，1删除',
  `ROLE_LASTMODIFY_TIME` datetime DEFAULT NULL,
  `ROLE_LASTMODIFY_ID` varchar(30) DEFAULT NULL,
  PRIMARY KEY (`ROLE_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `blog_cms_role_menu_r` (
  `RELATION_ID` varchar(30) NOT NULL,
  `ROLE_ID` varchar(30) DEFAULT NULL,
  `MENU_ID` varchar(30) DEFAULT NULL,
  PRIMARY KEY (`RELATION_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `blog_cms_user_info` (
  `USER_ID` varchar(30) NOT NULL,
  `OPERATION_DATA` longtext COMMENT '操作权限数据',
  PRIMARY KEY (`USER_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `blog_cms_user_operation` (
  `OPERATION_ID` varchar(30) NOT NULL,
  `OPERATION_NAME` varchar(30) DEFAULT NULL,
  `OPERATION_BRIEF` varchar(200) DEFAULT NULL,
  `OPERATION_TYPE` decimal(2,0) DEFAULT '0' COMMENT '0基本等级操作',
  `OPERATION_TAG` varchar(10) DEFAULT NULL,
  `CREATE_ID` varchar(30) DEFAULT NULL,
  `CREATE_TIME` timestamp NULL DEFAULT NULL,
  `DELETE_FLG` decimal(1,0) DEFAULT '0' COMMENT '0正常，1删除',
  `LASTMODIFY_ID` varchar(30) DEFAULT NULL,
  `LASTMODIFY_TIME` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`OPERATION_ID`),
  UNIQUE KEY `IDX_OPERATION_TAG` (`OPERATION_TAG`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


CREATE TABLE `blog_cms_user_role_r` (
  `USER_ID` varchar(30) NOT NULL,
  `ROLE_ID` varchar(30) NOT NULL,
  PRIMARY KEY (`USER_ID`,`ROLE_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
